class ToggleSwitch(QPushButton):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setCheckable(True)
        self.setMinimumWidth(50)
        self.setMaximumWidth(50)
        self.setMinimumHeight(24)
        self.setCursor(Qt.PointingHandCursor)
        self.update_style()
        self.toggled.connect(self.update_style)

    def update_style(self):
        if self.isChecked():
            self.setStyleSheet("""
                QPushButton {
                    background-color: #2d7dca;
                    border-radius: 12px;
                }
            """)
        else:
            self.setStyleSheet("""
                QPushButton {
                    background-color: #555;
                    border-radius: 12px;
                }
            """)

    def paintEvent(self, event):
        super().paintEvent(event)
        from PyQt5.QtGui import QPainter, QColor as _QColor

        painter = QPainter(self)
        painter.setRenderHint(QPainter.Antialiasing)

        radius = 20
        margin = 2

        if self.isChecked():
            x = self.width() - radius - margin
            color = _QColor("#ffffff")
        else:
            x = margin
            color = _QColor("#dddddd")

        painter.setBrush(color)
        painter.setPen(Qt.NoPen)
        painter.drawEllipse(x, margin, radius, radius)

    def _toggle_pwd_visibility(self):
        if self.edit_password.echoMode() == QLineEdit.Password:
            self.edit_password.setEchoMode(QLineEdit.Normal)
            self.btn_toggle_pwd.setText("üôà")
        else:
            self.edit_password.setEchoMode(QLineEdit.Password)
            self.btn_toggle_pwd.setText("üëÅÔ∏è")

    # --------------------------------------------------------
    #   GENERADOR DE CONTRASE√ëAS
    # --------------------------------------------------------
    def _generate_password(self):
        import secrets
        import string

        chars = string.ascii_letters + string.digits + "!@#$%^&*()-_=+"
        pwd = "".join(secrets.choice(chars) for _ in range(16))

        self.edit_password.setText(pwd)
        self._update_strength_meter()

    # --------------------------------------------------------
    #   MEDIDOR DE CALIDAD
    # --------------------------------------------------------
    def _update_strength_meter(self):
        pwd = self.edit_password.text()
        score = 0

        if len(pwd) >= 8: score += 20
        if len(pwd) >= 12: score += 20
        if any(c.islower() for c in pwd): score += 20
        if any(c.isupper() for c in pwd): score += 20
        if any(c.isdigit() for c in pwd): score += 10
        if any(c in "!@#$%^&*()-_=+" for c in pwd): score += 10

        self.strength_bar.setValue(score)

        if score < 40:
            color = "red"
            text = "D√©bil"
        elif score < 70:
            color = "orange"
            text = "Media"
        else:
            color = "green"
            text = "Fuerte"

        self.strength_bar.setStyleSheet(f"""
            QProgressBar {{
                border: 1px solid #ccc;
                border-radius: 4px;
                background: #eee;
            }}
            QProgressBar::chunk {{
                border-radius: 4px;
                background-color: {color};
            }}
        """)

        self.strength_label.setText(f"Fortaleza: {text}")

    # --------------------------------------------------------
    #   OBTENER DATOS
    # --------------------------------------------------------
    def get_data(self):
        return {
            "service": self.edit_service.text().strip(),
            "username": self.edit_user.text().strip(),
            "secret": self.edit_password.text().strip(),
            "notes": self.edit_notes.toPlainText().strip()
        }

    def check_supabase(self) -> bool:
        import requests
        try:
            url = self.sync_mgr.supabase_url + "/rest/v1/"
            r = requests.options(url, timeout=3)
            return r.status_code in (200, 204, 400, 401, 404)

    # ============================================================

    def _apply_table_filter(self, query: str):
        from PyQt5.QtGui import QColor as _QColor

        query = (query or "").lower()
        sorting_prev = self.table.isSortingEnabled()
        self.table.setSortingEnabled(False)
        self.table.clearContents()
        self.table.setRowCount(0)

        filtered = []
        for s in self._secrets_cache:
            if (not s.get("service")
                    and not s.get("username")
                    and not s.get("secret")):
                continue

            service = (s.get("service") or "").strip()
            user = (s.get("username") or "").strip()

            if query:
                if query not in service.lower() and query not in user.lower():
                    continue

            filtered.append(s)

        print(f"DEBUG: total en cache={len(self._secrets_cache)}, filtrados={len(filtered)}")

        self.table.setRowCount(len(filtered))

        for row_idx, s in enumerate(filtered):
            service = s.get("service", "")
            user = s.get("username", "")
            pwd = s.get("secret", "")
            notes = s.get("notes", "")
            sid = s.get("id")  # ID del registro, asumiendo que viene de la BD

            print(f"FILA {row_idx}: service={service!r}, user={user!r}, pwd_len={len(pwd)}")

            score = 0
            if len(pwd) >= 8:
                score += 25
            if any(c.islower() for c in pwd) and any(c.isupper() for c in pwd):
                score += 25
            if any(c.isdigit() for c in pwd):
                score += 25
            if any(c in "!@#$%^&*()-_=+[]{}|;:'\",.<>?/`~" for c in pwd):
                score += 25

            emoji = "üîí" if score >= 75 else "üîì"
            emoji_item = QTableWidgetItem(emoji)
            emoji_item.setTextAlignment(Qt.AlignCenter)
            emoji_item.setForeground(_QColor("black"))
            self.table.setItem(row_idx, 0, emoji_item)

            service_item = QTableWidgetItem(service)
            service_item.setForeground(_QColor("black"))
            self.table.setItem(row_idx, 1, service_item)

            user_item = QTableWidgetItem(user)
            user_item.setForeground(_QColor("black"))
            self.table.setItem(row_idx, 2, user_item)

            masked = "‚Ä¢" * len(pwd) if pwd else ""
            pwd_item = QTableWidgetItem(masked)
            pwd_item.setForeground(_QColor("black"))
            self.table.setItem(row_idx, 3, pwd_item)

            btn_copy = QPushButton("üìã")
            btn_copy.setCursor(Qt.PointingHandCursor)
            btn_copy.setStyleSheet("border: none; font-size: 16px;")
            btn_copy.clicked.connect(lambda _, p=pwd: self._copy_password(p))
            self.table.setCellWidget(row_idx, 4, btn_copy)

            btn_show = QPushButton("üëÅÔ∏è")
            btn_show.setCursor(Qt.PointingHandCursor)
            btn_show.setStyleSheet("border: none; font-size: 16px;")
            btn_show.clicked.connect(
                lambda _, r=row_idx, pw=pwd: self._toggle_row_password(r, pw)

            self.table.setCellWidget(row_idx, 5, btn_show)

    def _on_local_backup(self):
        from PyQt5.QtWidgets import QFileDialog

        try:
            # 1. Seleccionar carpeta destino
            target_dir = QFileDialog.getExistingDirectory(
                self,
                "Seleccionar carpeta para guardar backup"
            )

            if not target_dir:
                return  # Usuario cancel√≥

            target_path = Path(target_dir)

            # 2. Verificar que la unidad existe
            if not target_path.exists():
                QMessageBox.critical(
                    self,
                    "Unidad no disponible",
                    "La unidad o carpeta seleccionada no existe o no est√° disponible."
                )
                return

            # 3. Verificar permisos de escritura
            test_file = target_path / "test_write.tmp"
            try:
                with open(test_file, "w") as f:
                    f.write("test")
                test_file.unlink()
            except Exception:
                QMessageBox.critical(
                    self,
                    "Sin permisos",
                    "No se puede escribir en la carpeta seleccionada.\n"
                    "Seleccione otra ubicaci√≥n."
                )
                return

            # 4. Crear backup real
            backup_path = self.sm.create_local_backup()

            # 5. Mover backup al destino elegido
            backup_file = Path(backup_path)
            final_path = target_path / backup_file.name
            backup_file.replace(final_path)

            QMessageBox.information(
                self,
                "Backup local",
                f"Backup creado correctamente en:\n{final_path}"
            )

        except Exception as e:
            QMessageBox.critical(self, "Error", str(e))

    def _on_local_restore(self):
        from PyQt5.QtWidgets import QFileDialog
        from src.infrastructure.user_manager import UserManager

        try:
            # 1. Solicitar token TOTP
            dlg = TotpDialog(self)
            if dlg.exec_() != QDialog.Accepted:
                return

            token = dlg.get_token()
            um = UserManager()
            secret = um.get_or_create_totp_secret()

            if not um.verify_totp(secret, token):
                QMessageBox.critical(self, "Error", "Token 2FA incorrecto. Restauraci√≥n cancelada.")
                return

            # 2. Seleccionar archivo .enc
            file_path, _ = QFileDialog.getOpenFileName(
                self,
                "Seleccionar archivo de backup",
                "",
                "Backups cifrados (*.enc)"
            )

            if not file_path:
                return

            file_path = Path(file_path)

            if not file_path.exists():
                QMessageBox.critical(self, "Error", "El archivo seleccionado no existe.")
                return

            # 3. Leer archivo
            try:
                with open(file_path, "rb") as f:
                    data = f.read()
            except Exception:
                QMessageBox.critical(self, "Error", "No se puede leer el archivo seleccionado.")
                return

            # 4. Intentar descifrar
            try:
                nonce = data[:12]
                ciphertext = data[12:]

                from cryptography.hazmat.primitives.ciphers.aead import AESGCM
                aes = AESGCM(self.sm.master_key)

                decrypted = aes.decrypt(nonce, ciphertext, None)
                secrets_list = json.loads(decrypted.decode("utf-8"))

            except Exception:
                QMessageBox.critical(
                    self,
                    "Error",
                    "El archivo de backup est√° corrupto o la contrase√±a maestra es incorrecta."
                )
                return

            # 5. Restaurar en SQLite
            try:
                self.sm.conn.execute("DELETE FROM secrets")
                self.sm.conn.commit()

                for s in secrets_list:
                    self.sm.add_secret(
                        s.get("service"),
                        s.get("username"),
                        s.get("secret"),
                        s.get("notes")
                    )

                self._load_table()

                QMessageBox.information(
                    self,
                    "Restauraci√≥n completada",
                    "El backup local fue restaurado correctamente."
                )

            except Exception as e:
                QMessageBox.critical(self, "Error", f"No se pudo restaurar el backup:\n{e}")

        except Exception as e:
            QMessageBox.critical(self, "Error inesperado", str(e))

    def _generate_password_advanced(self):
        length = self.length_slider.value()

        upper = self.cb_upper.isChecked()
        lower = self.cb_lower.isChecked()
        digits = self.cb_digits.isChecked()
        symbols = self.cb_symbols.isChecked()

        if self.rb_max_entropy.isChecked():
            upper = lower = digits = symbols = True
            if length < 20:
                length = 20
                self.length_slider.setValue(20)
        elif self.rb_strict.isChecked():
            upper = lower = digits = True
            symbols = False
        elif self.rb_safe.isChecked():
            upper = lower = digits = True
            symbols = False

        alphabet = ""

        if self.rb_safe.isChecked():
            safe_lower = "abcdefghijkmnopqrstuvwxyz"
            safe_upper = "ABCDEFGHJKLMNPQRSTUVWXYZ"
            safe_digits = "23456789"
            if lower:
                alphabet += safe_lower
            if upper:
                alphabet += safe_upper
            if digits:
                alphabet += safe_digits
        else:
            if lower:
                alphabet += string.ascii_lowercase
            if upper:
                alphabet += string.ascii_uppercase
            if digits:
                alphabet += string.digits
            if symbols:
                alphabet += "!@#$%^&*()-_=+[]{}|;:'\",.<>?/`~"

        if not alphabet:
            QMessageBox.warning(
                self,
                "Generador",
                "Debes seleccionar al menos un tipo de car√°cter."
            )
            return

        pwd = ''.join(secrets.choice(alphabet) for _ in range(length))

        self.pass_input.setText(pwd)
        self._update_strength_bar(pwd)

        if self.pass_input.echoMode() == QLineEdit.Normal:
            self.toggle_pass_btn.setText("üôà")
        else:
            self.toggle_pass_btn.setText("üëÅÔ∏è")
